<script type="text/javascript">
	export default{
		props:{
			value:{},
			type:{
				type:String,
				default:'default'    //default 默认,danger 危险，confirm 确认
			},
			content:{
				type:String,
				default:''
			},
			title:{
				type:String,
				default:''
			},
			cancelText:{
				type:String,
				default:'取消'
			},
			dangerText:{
				type:String,
				default:'删除'
			},
			confirmText:{
				type:String,
				default:'确认'
			}
		},
		data(){
			return{
				showMask:false,
			}
		},
		methods:{
			closeMask(){
				this.showMask = false;
			},
			cancelBtn(){
				this.$emit('cancel');
				this.closeMask();
			},
			dangerBtn(){
				this.$emit('danger');
				this.closeMask();
			},
			confirmBtn(){
				this.$emit('confirm');
				this.closeMask();
			}
		},
		mounted(){
			this.showMask = this.value;
		},
		watch:{
			value(newVal,oldVal){
				this.showMask = newVal;
			},
			showMask(val){
				this.$emit('input',val);
			}
		}
	}
</script>
